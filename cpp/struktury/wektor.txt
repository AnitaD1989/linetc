struct punkt {
	int wx;
	int wy;
};

struct wektor {
	punkt start;
	punkt end;
};

void getWektor() {
	wektor w;
	w.start.wx=5;
	w.start.wy=3;
	w.end.wx=8;
	w.end.wy=9;
	srodek (w);
}

void srodek (wektor w) {
	cout << "x: " << (double)((w.start.wx+w.end.wx)/2) << "y: " << (double)((w.start.wy+w.end.wy)/2) << endl;
}

Napisz program, który pobiera od użytkownika współrzędne punktu początkowego i końcowego
wektora, a następnie oblicza współrzędne środka tego wektora.
